	\section{Revisão}	
	\subsection{Ordem de precedência}
	\begin{frame}{Ordem de precedência}
			De acordo com a definição de fórmula, o uso de parênteses é obrigatório ao se utilizar conectivos. Porém, iremos estabelecer algumas abreviações que serão permitidas:
		\begin{block}{}
			Os parênteses mais externos de uma fórmula podem ser omitidos.
			\begin{itemize}
				\item $\neg r \equiv (\neg r)$
				\item $p \wedge q \equiv (p \wedge q)$
				\item $(r \wedge \neg q) \rightarrow \neg p \equiv ((r \wedge \neg q) \rightarrow \neg p)$
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Ordem de precedência}
			De acordo com a definição de fórmula, o uso de parênteses é obrigatório ao se utilizar conectivos. Porém, iremos estabelecer algumas abreviações que serão permitidas:
		\begin{block}{}
			O uso repetido dos conectivos $\wedge$ e $\vee$ dispensa o uso dos parênteses. 
			
			\begin{itemize}
				\item $p \wedge q \wedge \neg r \wedge \neg s \equiv ((p \wedge q) \wedge \neg r) \wedge \neg s$
				\item $r \vee s \wedge \neg t \vee p \equiv ((r \vee s) \wedge \neg t) \vee p$
			\end{itemize}
		\end{block}
		\begin{alertblock}{Observação}
			Note que os parênteses aninham-se {\bf à esquerda}.
		\end{alertblock}

	\end{frame}
	
	\begin{frame}{Ordem de precedência}
			De acordo com a definição de fórmula, o uso de parênteses é obrigatório ao se utilizar conectivos. Porém, iremos estabelecer algumas abreviações que serão permitidas:
		\begin{block}{}
			O uso repetido do conectivo $\rightarrow$ também dispensa o uso dos parênteses.
			\begin{itemize}
				\item $p \rightarrow q \rightarrow \neg r \equiv p \rightarrow (q \rightarrow \neg r)$
				\item $r \rightarrow \neg s \rightarrow \neg t \equiv r \rightarrow (\neg s \rightarrow \neg t)$
			\end{itemize}
		\end{block}		
		\begin{alertblock}{Observação}
			Note que os parênteses aninham-se {\bf à direita}.
		\end{alertblock}
	\end{frame}
	
	\begin{frame}{Ordem de precedência}
			De acordo com a definição de fórmula, o uso de parênteses é obrigatório ao se utilizar conectivos. Porém, iremos estabelecer algumas abreviações que serão permitidas:
		\begin{block}{}
			Quando houver uma combinação de conectivos, existe uma precedência entre eles, dada pela ordem: $\neg, \wedge, \vee, \rightarrow$.			
			\begin{itemize}
				\item $\neg p \wedge q \equiv ((\neg p) \wedge q)$
				\item $p \vee q \wedge r \equiv p \vee (q \wedge r)$
				\item $p \vee \neg q \rightarrow r \equiv (p \vee (\neg q)) \rightarrow r$
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Ordem de precedência}
			De acordo com a definição de fórmula, o uso de parênteses é obrigatório ao se utilizar conectivos. Porém, iremos estabelecer algumas abreviações que serão permitidas:
		\begin{alertblock}{Recomendação}
			Em geral, deve-se preferir clareza à economia de parênteses e, na dúvida, é bom deixar alguns parênteses para explicitar o sentido da fórmula.
		\end{alertblock}
	\end{frame}
	
	\section{Tamanho das Fórmulas}
	\begin{frame}{Tamanho das Fórmulas}
		\begin{block}{Tamanho das Fórmulas}
			O tamanho ou complexidade de uma fórmula A, representado por $|A|$, é um numéro inteiro definido como se segue:
			\pause
			\begin{enumerate}
				\item<2,5> {\bf Caso básico:} $|p|=1$ \\para toda fórmula atômica $p \in \mathcal{P}$;
				\item<3,5> {\bf Caso} $|(\neg A)|$ \\$|(\neg A)| = 1 + |A|$.
				\item<4,5> {\bf Caso} $|(A \circ B)|$ \\$|(A \circ B)| = 1 + |A| + |B|$, para $\circ \in \{ \wedge, \vee, \rightarrow \}$.
			\end{enumerate}
		\end{block}
	\end{frame}
	
	\section{Semântica da LP}
	\begin{frame}{Semântica da LP}
		\begin{block}{Semântica}
			O estudo da semântica da lógica proposicional consiste em atribuir {\it valores verdade} às fórmulas da linguagem. Na lógica clássica, há apenas dois valores verdade: {\it verdadeiro} e {\it falso}. Representaremos o {\it verdadeiro} por $1$ e o {\it falso} por $0$.
		\end{block}
		\pause
		\begin{block}{Função de Valoração $\cal{V}$}
			$\mathcal{V} : \mathcal{P} \rightarrow \{ 0,1 \}$
		\end{block}
	\end{frame}
	
	\begin{frame}{Semântica da LP}
		\begin{block}{Valoração de uma fórmula qualquer}
			\begin{enumerate}
				\item<2,7> $\cal{V} : \cal{P} \rightarrow \{$ $0,1$ $\}$ ({\bf Caso básico}).				
				\item<3,7> $\mathcal{V}(\neg A) = 1$ \\se, e somente se, $\mathcal{V}(A) = 0$.
				\item<4,7> $\mathcal{V}(A \wedge B) = 1$ \\se, e somente se, $\mathcal{V}(A) = 1$ e $\mathcal{V}(B) = 1$.
				\item<5,7> $\mathcal{V}(A \vee B) = 1$ \\sse $\mathcal{V}(A) = 1$ ou $\mathcal{V}(B) = 1$.
				\item<6,7> $\mathcal{V}(A \rightarrow B) = 1$ \\sse $\mathcal{V}(A) = 0$ ou $\mathcal{V}(B) = 1$.
			\end{enumerate}
		\end{block}
	\end{frame}
	
	\begin{frame}{Matriz de Conectivos Lógicos}
		\begin{block}{Conectivo $\neg$}
			\begin{tabular}{c|c}
				        & $\neg A$  \\ \hline
				$A = 0$ & 1  \\
				$A = 1$ & 0 
			\end{tabular}
		\end{block}
		\pause
		\begin{block}{Conectivo $\wedge$}
			\begin{tabular}{c|cc}
				$A \wedge B$ & $B = 0$ & $B = 1$ \\ \hline
				$A = 0$ & 0 & 0 \\
				$A = 1$ & 0 & 1
			\end{tabular}
		\end{block}
	\end{frame}
	
	\begin{frame}{Matriz de Conectivos Lógicos}
		\begin{block}{Conectivo $\vee$}
			\begin{tabular}{c|cc}
				$A \vee B$ & $B = 0$ & $B = 1$ \\ \hline
				$A = 0$ & 0 & 1 \\
				$A = 1$ & 1 & 1
			\end{tabular}
		\end{block}		
		\pause
		\begin{block}{Conectivo $\rightarrow$}
			\begin{tabular}{c|cc}
				$A \rightarrow B$ & $B = 0$ & $B = 1$ \\ \hline
				$A = 0$ & 1 & 1 \\
				$A = 1$ & 0 & 1
			\end{tabular}
		\end{block}
	\end{frame}
	
	\begin{frame}{Valoração de Fórmulas}
		Dada a fórmula $A = (p \vee \neg q) \rightarrow (r \wedge \neg q)$ \pause
		\begin{block}{$\mathcal{V}_1(A)$}
			Em que temos $\mathcal{V}_1(p) = 1$, $\mathcal{V}_1(q) = 0$ e $\mathcal{V}_1(r) = 1$. \pause
			\\ \begin{center} \shadowbox{$\mathcal{V}_1(A) = 1$} \end{center}
		\end{block}
		\pause
		\begin{block}{$\mathcal{V}_2(A)$}
			Em que temos $\mathcal{V}_2(p) = 1$, $\mathcal{V}_2(q) = 1$ e $\mathcal{V}_2(r) = 1$. \pause
			\\ \begin{center} \shadowbox{$\mathcal{V}_2(A) = 0$} \end{center}
		\end{block}
	\end{frame}
	
	\begin{frame}{Valoração de Fórmulas}
		\begin{block}{Possibilidades de valorações diferentes}
			Se uma fórmula $A$ possui $N$ subfórmulas atômicas, e cada valoração pode atribuir ou 0 ou 1 a cada um desses átomos, temos que pode haver $2^N$ distintas valorações para a fórmula $A$.
		\end{block}
	\end{frame}
