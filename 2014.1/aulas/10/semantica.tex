	\section{Revisão}	
	\subsection{Semântica da LP}
	\begin{frame}{Semântica da LP}
		\begin{block}{Semântica}
			O estudo da semântica da lógica proposicional consiste em atribuir {\it valores verdade} às fórmulas da linguagem. Na lógica clássica, há apenas dois valores verdade: {\it verdadeiro} e {\it falso}. Representaremos o {\it verdadeiro} por $1$ e o {\it falso} por $0$.
		\end{block}
		\begin{block}{Função de Valoração $\cal{V}$}
			$\mathcal{V} : \mathcal{P} \rightarrow \{ 0,1 \}$
		\end{block}
	\end{frame}
	
	\begin{frame}{Semântica da LP}
		\begin{block}{Valoração de uma fórmula qualquer}
			\begin{enumerate}
				\item $\cal{V} : \cal{P} \rightarrow \{$ $0,1$ $\}$ ({\bf Caso básico}).				
				\item $\mathcal{V}(\neg A) = 1$ \\se, e somente se, $\mathcal{V}(A) = 0$.
				\item $\mathcal{V}(A \wedge B) = 1$ \\se, e somente se, $\mathcal{V}(A) = 1$ e $\mathcal{V}(B) = 1$.
				\item $\mathcal{V}(A \vee B) = 1$ \\sse $\mathcal{V}(A) = 1$ ou $\mathcal{V}(B) = 1$.
				\item $\mathcal{V}(A \rightarrow B) = 1$ \\sse $\mathcal{V}(A) = 0$ ou $\mathcal{V}(B) = 1$.
			\end{enumerate}
		\end{block}
	\end{frame}
	
	\begin{frame}{Matriz de Conectivos Lógicos}
		\begin{block}{Conectivo $\neg$}
			\begin{tabular}{c|c}
				        & $\neg A$  \\ \hline
				$A = 0$ & 1  \\
				$A = 1$ & 0 
			\end{tabular}
		\end{block}
		
		\begin{block}{Conectivo $\wedge$}
			\begin{tabular}{c|cc}
				$A \wedge B$ & $B = 0$ & $B = 1$ \\ \hline
				$A = 0$ & 0 & 0 \\
				$A = 1$ & 0 & 1
			\end{tabular}
		\end{block}
	\end{frame}
	
	\begin{frame}{Matriz de Conectivos Lógicos}
		\begin{block}{Conectivo $\vee$}
			\begin{tabular}{c|cc}
				$A \vee B$ & $B = 0$ & $B = 1$ \\ \hline
				$A = 0$ & 0 & 1 \\
				$A = 1$ & 1 & 1
			\end{tabular}
		\end{block}
		\begin{block}{Conectivo $\rightarrow$}
			\begin{tabular}{c|cc}
				$A \rightarrow B$ & $B = 0$ & $B = 1$ \\ \hline
				$A = 0$ & 1 & 1 \\
				$A = 1$ & 0 & 1
			\end{tabular}
		\end{block}
	\end{frame}
	
	\begin{frame}{Valoração de Fórmulas}
		Dada a fórmula $A = (p \vee \neg q) \rightarrow (r \wedge \neg q)$ 
		\begin{block}{$\mathcal{V}_1(A)$}
			Em que temos $\mathcal{V}_1(p) = 1$, $\mathcal{V}_1(q) = 0$ e $\mathcal{V}_1(r) = 1$. 
			\\ \begin{center} \shadowbox{$\mathcal{V}_1(A) = 1$} \end{center}
		\end{block}
		\pause
		\begin{block}{$\mathcal{V}_2(A)$}
			Em que temos $\mathcal{V}_2(p) = 1$, $\mathcal{V}_2(q) = 1$ e $\mathcal{V}_2(r) = 1$. \pause
			\\ \begin{center} \shadowbox{$\mathcal{V}_2(A) = 0$} \end{center}
		\end{block}
	\end{frame}
	
	\begin{frame}{Valoração de Fórmulas}
		\begin{block}{Possibilidades de valorações diferentes}
			Se uma fórmula $A$ possui $N$ subfórmulas atômicas, e cada valoração pode atribuir ou 0 ou 1 a cada um desses átomos, temos que pode haver $2^N$ distintas valorações para a fórmula $A$.
		\end{block}
	\end{frame}
	
	\section{Satisfazibilidade}
	\begin{frame}{Satisfazibilidade e Validade}
		\begin{block}{Satisfazibilidade}
			Uma fórmula $A$ é dita {\it satisfazível} se existe uma valoração $\mathcal{V}$ de seus átomos tal que $\mathcal{V}(A) = 1$.
		\end{block} \pause
		\begin{block}{Insatisfazilidade}
			Uma fórmula $A$ é dita {\it insatisfazível} se toda valoração $\mathcal{V}$ de seus átomos é tal que $\mathcal{V}(A) = 0$.
		\end{block}
	\end{frame}	
	
	\begin{frame}{Validade}
		\begin{block}{Validade}
			Uma fórmula $A$ é dita {\it válida} ou uma {\it tautologia} se toda valoração $\mathcal{V}$ de seus átomos é tal que $\mathcal{V}(A) = 1$.
		\end{block} \pause
		\begin{block}{Falsificabilidade}
			Uma fórmula $A$ é dita {\it falsificável} se existe uma valoração $\mathcal{V}$ de seus átomos tal que $\mathcal{V}(A) = 0$.
		\end{block}
	\end{frame}
	
	\begin{frame}{Corolários}
		\begin{block}{Corolário 01}
			Toda fórmula válida é também satisfazível.
		\end{block} \pause
		\begin{block}{Corolário 02}
			Toda fórmula insatisfazível é falsificável.
		\end{block} \pause
		\begin{block}{Corolário 03}
			Uma fórmula não pode ser satisfazível e insatisfazível.
		\end{block} \pause
		\begin{block}{Corolário 04}
			Uma fórmula não pode ser válida e falsificável.
		\end{block}
	\end{frame}
	
	\begin{frame}{Corolários}
		\begin{block}{Corolário 05}
			Se $A$ é válida, então $\neg A$ é insatisfatível; analogamente se $A$ é insatisfatível, entao $\neg A$ é válida.
		\end{block} \pause
		\begin{block}{Corolário 06}
			Se $A$ é satisfatível, $\neg A$ é falsificável, e vice-versa.
		\end{block} \pause
		\begin{block}{Corolário 07}
			Existem fórmulas que são tanto satisfatíveis como falsificáveis.
		\end{block} 
	\end{frame}	
	
	\begin{frame}{Desafio na Computação}
		\begin{block}{Problema em Aberto}
			Dada uma fórmula complexa qualquer, classificá-la como 
			\begin{itemize}
				\item satisfazível;
				\item insatisfazível;
				\item válida;
				\item falsificável.
			\end{itemize}
		\end{block} \pause
		\begin{block}{Primeira solução...}
			Construção de tabela da verdade.
		\end{block}
	\end{frame}
