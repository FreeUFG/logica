	\section{Tabela da Verdade}
	\begin{frame}{Tabela da Verdade}
		\begin{block}{}
			Foi um dos primeiros métodos propostos na literatura para a verificação da satisfazibilidade e validade de fórmulas.
		\end{block}
		\begin{block}{Definição}
			A Tabela da Verdade é um método exaustivo de geração de valorações para uma dada fórmula $A$.
		\end{block}
	\end{frame}
	
	\begin{frame}{Tabela da Verdade}
		\begin{block}{Construção}
			\begin{itemize}
				\item<1,5> A tabela possui uma coluna para cada subfórmula de $A$, inclusive para $A$. Em geral, os átomos de $A$ ficam situados nas colunas mais à esquerda, e $A$ é a fórmula mais à direita.
				\item<2,5> Para cada valoração possível para os átomos de $A$, insere-se uma linha com os valores da valoração dos átomos.
				\item<3,5> Em seguida, a valoração dos átomos é propagada para as subfórmulas, obedecendo-se a definição de valoração. Dessa forma, começa-se valorando as fórmulas menores até as maiores.
				\item<4,5> Ao final desse processo, todas as possíveis valorações de $A$ são criadas.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Tabela da Verdade}
		\begin{block}{Classificação}
			\begin{itemize}
				\item<1,5> $A$ é satisfazível se alguma linha da coluna $A$ contiver 1;
				\item<2,5> $A$ é válida se todas as linhas da coluna $A$ contiverem 1;
				\item<3,5> $A$ é falsificável se alguma linha da coluna $A$ contiver 0;
				\item<4,5> $A$ é insatisfazível se todas as linhas da coluna $A$ contiverem 0.
			\end{itemize}
		\end{block}
	\end{frame}

	\section{Consequência Lógica}
	\begin{frame}{Consequência Lógica}
		\begin{block}{$A \models B$}
			Dizemos que $B$ é {\it consequência lógica} de $A$ (representada por $A \models B$) se toda valoração $\mathcal{V}$ que satisfaz $A$ também satisfaz $B$.
		\end{block} \pause
		\begin{block}{$A \models B$}
			Lê-se $B$ é consequência lógica de $A$, ou $A$ {\it implica logicamente} $B$.
		\end{block}
	\end{frame}
	
	\begin{frame}{Consequência Lógica}
		\begin{block}{Verificar se...}
			$p \vee q \rightarrow r \models p \rightarrow r$ \mbox{   } \pause {\color{green} \bf  V}
		\end{block} \pause
		\begin{block}{Verificar se...}
			$p \wedge q \rightarrow r \models p \rightarrow r$ \mbox{   } \pause {\color{red} \bf F}
		\end{block}\pause
		\begin{block}{Logo...}
			$p \wedge q \rightarrow r \not\models p \rightarrow r$
		\end{block}
	\end{frame}
