	\section{Revisão}	
	\subsection{Consequência Lógica}
	\begin{frame}{Consequência Lógica}
		\begin{block}{$A \models B$}
			Dizemos que $B$ é {\it consequência lógica} de $A$ (representada por $A \models B$) se toda valoração $\mathcal{V}$ que satisfaz $A$ também satisfaz $B$.
		\end{block}
		\begin{block}{$A \models B$}
			Lê-se $B$ é consequência lógica de $A$, ou $A$ {\it implica logicamente} $B$.
		\end{block}
	\end{frame}
	
	\begin{frame}{Consequência Lógica}
		\begin{block}{Verificar se...}
			$p \vee q \rightarrow r \models p \rightarrow r$ \mbox{   } \pause {\color{green} \bf  V}
		\end{block}
		\begin{block}{Verificar se...}
			$p \wedge q \rightarrow r \models p \rightarrow r$ \mbox{   } \pause {\color{red} \bf F}
		\end{block}\pause
		\begin{block}{Logo...}
			$p \wedge q \rightarrow r \not\models p \rightarrow r$
		\end{block}
	\end{frame}
	
	\section{Consequência Lógica}
	\begin{frame}{Consequência Lógica}
		\begin{block}{Eliminação da dupla negação}
			$\neg \neg p \models p $ 
		\end{block} \pause
		\begin{block}{Introdução da dupla negação}
			$p \models \neg \neg p $  
		\end{block}	\pause			
		\begin{block}{Eliminação do $\wedge$}
			$p \wedge q \models p $\\
			$p \wedge q \models q $ 
		\end{block} \pause
		\begin{block}{Adição do $\vee$}
			$p \models p \vee q $
		\end{block}
	\end{frame}
	
	\begin{frame}{Consequência Lógica}
		\begin{block}{\it Modus Ponens}
			$(p \rightarrow q) \wedge p \models q $
		\end{block} \pause
		\begin{block}{\it Modus Tollens}
			$(p \rightarrow q) \wedge \neg q \models \neg p$
		\end{block} \pause
		\begin{block}{Redução ao absurdo}
			$(p \rightarrow q) \wedge (p \rightarrow \neg q) \models \neg p $
		\end{block} \pause
		\begin{block}{Silogismo Disjuntivo}
			$(p \vee q) \wedge \neg p  \models q $
		\end{block}
	\end{frame}
