	\section{Revisão}	
	\begin{frame}{Consequência Lógica}
		\begin{block}{Eliminação da dupla negação}
			$\neg \neg p \models p $ 
		\end{block}
		\begin{block}{Introdução da dupla negação}
			$p \models \neg \neg p $  
		\end{block}			
		\begin{block}{Eliminação do $\wedge$}
			$p \wedge q \models p $\\
			$p \wedge q \models q $ 
		\end{block} \pause
		\begin{block}{Adição do $\vee$}
			$p \models p \vee q $
		\end{block}
	\end{frame}
	
	\section{Consequência Lógica (Cont.)}
	\begin{frame}{Consequência Lógica}
		\begin{block}{\it Modus Ponens}
			$(p \rightarrow q) \wedge p \models q $
		\end{block} \pause
		\begin{block}{\it Modus Tollens}
			$(p \rightarrow q) \wedge \neg q \models \neg p$
		\end{block} \pause
		\begin{block}{Redução ao absurdo}
			$(p \rightarrow q) \wedge (p \rightarrow \neg q) \models \neg p $
		\end{block} \pause
		\begin{block}{Silogismo Disjuntivo}
			$(p \vee q) \wedge \neg p  \models q $
		\end{block}
	\end{frame}
	
	\section{Argumentação}
	\begin{frame}{Argumento}
		\begin{block}{Argumento}
			Um argumento pode ser representado em forma simbólica como
			
			$p_1 \wedge p_2 \wedge p_3 \wedge \cdots \wedge p_n \models q$ \pause
			
			em que $p_1, p_2, p_3 \cdots, p_n$ e $q$ são fórmulas proposicionais.
		\end{block} \pause
		\begin{block}{Premissas}
			Chamamos $p_1, p_2, p_3, \cdots, p_n$ de {\it premissas} (ou {\it hipóteses}) do argumento.
		\end{block} \pause
		\begin{block}{Conclusão}
			Chamamos $q$ de {\it conclusão} do argumento.
		\end{block}
	\end{frame}
	
	\begin{frame}{Argumento}
		\begin{block}{Terminologia}
			\begin{itemize}
				\item<2,7> $q$ é uma {\it consequência lógica} de $p_1, p_2, p_3, \cdots, p_n$
				\item<3,7> $q$ pode ser {\it deduzido logicamente} de $p_1, p_2, p_3, \cdots, p_n$
				\item<4,7> $q$ é uma {\it conclusão lógica} de $p_1, p_2, p_3, \cdots, p_n$
				\item<5,7> $p_1, p_2, p_3, \cdots, p_n$ {\it implica logicamente} em $q$
				\item<6,7> $q$ {\it segue logicamente} de $p_1, p_2, p_3, \cdots, p_n$
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Argumento}
		\begin{block}{Argumento Válido}
			Um argumento é válido se $p_1 \wedge p_2 \wedge p_3 \wedge \cdots \wedge p_n \models q$ for válida.
		\end{block} \pause
		\begin{block}{Exemplos}
			\begin{itemize}
				\item<3,7> $p \wedge q \models r$ é um argumento válido?
				\item<4,7> $p \models p \vee q$ é um argumento válido?
				\item<5,7> $(p \rightarrow q) \wedge p \models q$ é um argumento válido?
				\item<6,7> $(p \rightarrow q) \wedge q \models p$ é um argumento válido?
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Argumento}
		\begin{alert}{Problema}
			É possível garantir a validade de um argumento sem ter que recorrer à construção de uma tabela-verdade?
		\end{alert} \pause
		\begin{block}{Exemplos}
			\begin{itemize}
				\item<3,6> $(p \rightarrow q) \wedge (p \wedge r)\models q$ é um argumento válido?
				\item<4,6> $(p \wedge q) \wedge ((p \vee r) \rightarrow s)) \models p \wedge s $ é um argumento válido?
				\item<5,6> $(p \rightarrow (q \rightarrow r)) \wedge (p \rightarrow q) \wedge p \models r$ é um argumento válido?
			\end{itemize}
		\end{block}
	\end{frame}
