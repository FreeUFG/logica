	\section{Argumentação}
	\begin{frame}{Regras de Inferência}
		\begin{block}{{\it Modus Ponens} ($\rightarrow e$ ou MP)}
			\begin{eqnarray*}
				(1) & A \rightarrow B & \\
				(2) & A & \\
				 & \line(1,0){25} & \\
				(3) & B & \hspace{1cm} MP \mbox{ } (1),(2)
			\end{eqnarray*}
		\end{block} \pause
		\begin{block}{Expressão Lógica}
			$(A \rightarrow B) \wedge A \models B$
		\end{block}
	\end{frame}
	
	\begin{frame}{Regras de Inferência}
		\begin{block}{{\it Modus Tollens} (MT)}
			\begin{eqnarray*}
				(1) & A \rightarrow B & \\
				(2) & \neg B & \\
				 & \line(1,0){25} & \\
				(3) & \neg A & \hspace{1cm} MT \mbox{ } (1),(2)
			\end{eqnarray*}
		\end{block} \pause
		\begin{block}{Expressão Lógica}
			$(A \rightarrow B) \wedge \neg B \models \neg A$
		\end{block}
	\end{frame}
	
	\begin{frame}{Regras de Inferência}
		\begin{block}{Silogismo Hipotético ($SH$)}
			\begin{eqnarray*}
				(1) & A \rightarrow B& \\
				(2) & B \rightarrow C& \\
				 & \line(1,0){25} & \\
				(3) & A \rightarrow C & \hspace{1cm} SH \mbox{ } (1),(2)
			\end{eqnarray*}
		\end{block} \pause
		\begin{block}{Expressão Lógica}
			$(A \rightarrow B) \wedge (B \rightarrow C) \models A \rightarrow C$
		\end{block}
	\end{frame}
	
	\begin{frame}{Regras de Inferência}
		\begin{block}{Contradição ($\neg e$)}
			\begin{eqnarray*}
				(1) & A & \\
				(2) & \neg A & \\
				 & \line(1,0){25} & \\
				(3) & \perp & \hspace{1cm} \neg e \mbox{ } (1),(2)
			\end{eqnarray*}
		\end{block} \pause
		\begin{block}{Expressão Lógica}
			$A \wedge \neg A \models \perp$
		\end{block}
	\end{frame}
	
	\begin{frame}{Regras de Inferência}
		\begin{block}{Introdução da Negação ($\neg i$)}
			\begin{eqnarray*}
				(1) & A & \\
				(2) & \perp & \\
				 & \line(1,0){25} & \\
				(3) & \neg A & \hspace{1cm} \neg i \mbox{ } (1),(2)
			\end{eqnarray*}
		\end{block} \pause
		\begin{block}{Expressão Lógica}
			$A \wedge \perp \models \neg A$
		\end{block}
	\end{frame}
	
	\begin{frame}{Regras de Inferência}
		\begin{block}{De Morgan ($DM_\vee$)}
			\begin{eqnarray*}
				(1) & \neg (A \vee B) & \\
				 & \line(1,0){25} & \\
				(2) & \neg A \wedge \neg B & \hspace{1cm} DM_\vee \mbox{ } (1)
			\end{eqnarray*}
		\end{block} \pause
		\begin{block}{Expressão Lógica}
			$\neg (A \vee B) \models \neg A \wedge \neg B$
		\end{block}
	\end{frame}
	
	\begin{frame}{Regras de Inferência}
		\begin{block}{De Morgan ($DM_\wedge$)}
			\begin{eqnarray*}
				(1) & \neg (A \wedge B) & \\
				 & \line(1,0){25} & \\
				(2) & \neg A \vee \neg B & \hspace{1cm} DM_\wedge \mbox{ } (1)
			\end{eqnarray*}
		\end{block} \pause
		\begin{block}{Expressão Lógica}
			$\neg (A \wedge B) \models \neg A \vee \neg B$
		\end{block}
	\end{frame}

	\begin{frame}{Demonstração por Absurdo}

			Utilizada como uma alternativa à demonstração direta. Consiste em apresentar um absurdo diante da negação da conclusão como premissa.
		\end{block}
	\end{frame}
